#include <stdio.h>

// int型を返すmain関数の宣言
int main() {
    // printf関数. ""で囲まれた文字列を端末に出力する.
    // \nは改行コードを意味する.
    printf("DDD\nD  D\nD   D\nD    D\nD   D\nD  D\nDDD\n");

    // main関数の正常終了
    return 0;
}
